// 创建或打开文件内核对象;
HANDLE fileH = CreateFile("data.txt",
    GENERIC_READ|GENERIC_WRITE,
    FILE_SHARE_READ,
    NULL,
    OPEN_EXISTING,
    FILE_ATTRIBUTE_NORMAL,
    NULL);
if(fileH == INVALID_HANDLE_VALUE)
{
    cout<<"error in CreateFile"<<endl;
    return -1;
}

// 创建一个文件映射内核对象;
HANDLE mapFileH = CreateFileMapping(fileH,
    NULL,
    PAGE_READWRITE,
    0,
    0,
    "Resource");
if(mapFileH == NULL)
{
    cout<<"error in CreateFileMapping"<<endl;
    return -1;
}

// 将文件数据映射到进程的地址空间;
char * mapH = (char *)MapViewOfFile( mapFileH,
    FILE_MAP_ALL_ACCESS,
    0,
    0,
    0);
if(mapH == NULL)
{
    cout<<"error in MapViewOfFile"<<endl;
    return -1;
}

// 读取数据;
char *buf = mapH;
double k;
int i = 1;
while(1)
{
    k = atof(buf);
    cout << i << ": " << k << endl;
    if (i % 3 == 0)
    {
        buf = strstr(buf, "\n");  
    }
    else
    {
        buf = strstr(buf, " ");
    }
    if (strstr(buf, " ") == NULL)
    {
        cout << i << endl;
        break;
    }
    i++;
}

// 关闭句柄;
UnmapViewOfFile(mapH);
CloseHandle(mapFileH);
CloseHandle(fileH);
